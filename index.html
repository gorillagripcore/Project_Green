exercise1.ps1
Add-Type -AssemblyName System.Windows.Forms $radius = 100 # pixels from center $steps = 60 # points per full rotation $delay = 30 # ms per step (~1.8s per loop) Write-Host "Exercise 1: cursor animation running. Press Ctrl+C to stop." Write-Host "Moving in a circle (radius: $radius px)`n" $origin = [System.Windows.Forms.Cursor]::Position $i = 0 while ($true) { $angle = 2 * [Math]::PI * $i / $steps $x = [int]($origin.X + $radius * [Math]::Cos($angle)) $y = [int]($origin.Y + $radius * [Math]::Sin($angle)) [System.Windows.Forms.Cursor]::Position = [System.Drawing.Point]::new($x, $y) Start-Sleep -Milliseconds $delay $i = ($i + 1) % $steps if ($i -eq 0) { $time = Get-Date -Format "HH:mm:ss" Write-Host "[$time] Loop" } }
